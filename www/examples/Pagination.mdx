## Pagination

```jsx live=true
const Example = () => {
  const [activePage, setActivePage] = React.useState(1);
  const onSelect = (pageNumber) => setActivePage(pageNumber);

  const props = {
    pageCount: 10,
    activePage,
    onSelect,
    next: true,
    prev: true,
  };

  return (
    <div style={{ width: '300px' }}>
      <p>Current page is {activePage}</p>
      <Pagination {...props} />
    </div>
  );
};

render(Example);
```
