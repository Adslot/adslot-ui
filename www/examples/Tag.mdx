## Tag

```jsx live=true
const initialState = [
  {
    text: 'Display',
    accent: 'positive',
    id: '0ac7d4ce-af36-4fd9-b142-6377f8ad5f17',
    actionIconSvgHref: './assets/svg-symbols.svg#cancel',
  },
  {
    text: 'Mobile',
    accent: 'pending',
    id: '5a884b04-223b-49d9-91d4-7abf10ea608f',
    actionIconSvgHref: './assets/svg-symbols.svg#cancel',
  },
  {
    text: 'Run of Network',
    accent: 'negative',
    id: '87ecaf18-bf7f-472e-86e6-c91dd2700b78',
    actionIconSvgHref: './assets/svg-symbols.svg#cancel',
  },
];

const Example = () => {
  const [tags, setTags] = React.useState(initialState);

  function resetTags() {
    setTags(initialState);
  }

  function deleteTag(tagIdToDelete) {
    setTags(_.reject(tags, tag => tag.id === tagIdToDelete));
  }

  if (_.isEmpty(tags)) {
    return (
      <Button className="btn-inverse" onClick={resetTags}>
        Reset tag example
      </Button>
    );
  }

  return (
    <div>
      {_.map(tags, tag => (
        <Tag
          key={tag.id}
          id={tag.id}
          accent={tag.accent}
          baseClass={tag.baseClass}
          onAction={deleteTag}
          inverse
          actionIcon={<SvgSymbol href={tag.actionIconSvgHref} />}
        >
          {tag.text}
        </Tag>
      ))}
    </div>
  );
};

render(Example);
```
