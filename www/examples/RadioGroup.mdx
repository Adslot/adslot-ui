## Radio Group

```jsx live=true
const Example = () => {
  const [hobbies, setHobbies] = React.useState('badminton');
  const handleGroupChange = (value) => setHobbies(value);

  return (
    <>
      <RadioGroup name="hobbies" value={hobbies} onChange={handleGroupChange} dts="radio-group-dts">
        <Radio value="swimming" label="Swimming" dts="radio-dts" />
        <Radio value="soccer" label="Soccer" />
        <Radio value="badminton" label="Badminton" />
      </RadioGroup>
      <br />
      <h4>Horizontal RadioGroup</h4>
      <RadioGroup
        orientation="horizontal"
        name="hobbies"
        value={hobbies}
        onChange={handleGroupChange}
        dts="radio-group-dts"
      >
        <Radio id="r1" value="swimming" label="Swimming" dts="radio-dts" />
        <Radio id="r2" value="soccer" label="Soccer" />
        <Radio id="r3" value="badminton" label="Badminton" />
      </RadioGroup>
    </>
  );
};

render(Example);
```
