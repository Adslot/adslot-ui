## Search

```jsx live=true
const Example = () => {
  const [value, setValue] = React.useState('');
  const [searchOnInputChangeLoading, setSearchOnInputChangeLoading] = React.useState(false);
  const [searchOnEnterKeyLoading, setSearchOnEnterKeyLoading] = React.useState(false);
  const [searchInput, showSearchInput] = React.useState(false);

  const onChange = value => {
    setValue(value);
  };

  const onBlur = () => {
    showSearchInput(false);
  };

  const searchOnInputChange = () => {
    setSearchOnInputChangeLoading(true);
    setTimeout(() => setSearchOnInputChangeLoading(false), 950);
  };

  const searchOnEnterKey = value => {
    setSearchOnEnterKeyLoading(true);
    setTimeout(() => {
      setSearchOnEnterKeyLoading(false);
      alert(`You are searching for '${value}'`);
    }, 950);
  };

  return (
    <div style={{ width: '400px' }}>
      Search on input Changed
      <Search
        placeholder="Search on Change"
        value={value}
        onChange={onChange}
        onSearch={searchOnInputChange}
        debounceInterval={500}
        isLoading={searchOnInputChangeLoading}
      />
      <br />
      Search on Enter key pressed
      <Search
        placeholder="Search on ENTER"
        isLoading={searchOnEnterKeyLoading}
        onSearch={searchOnEnterKey}
        searchOnEnter
      />
      <br />
      Search on Enter key pressed without a button
      <Search
        placeholder="Search on ENTER"
        isLoading={searchOnEnterKeyLoading}
        onSearch={searchOnEnterKey}
        searchOnEnter
        showSearchButton={false}
      />
      <br />
      Expandable Search Bar
      {searchInput ? (
        <Search
          placeholder="Search on ENTER"
          isLoading={searchOnEnterKeyLoading}
          onSearch={searchOnEnterKey}
          onBlur={onBlur}
          searchOnEnter
        />
      ) : (
        <Button className="btn btn-borderless" onClick={() => showSearchInput(true)}>
          <SvgSymbol href="./assets/svg-symbols.svg#search" />
        </Button>
      )}
    </div>
  );
};

render(Example);
```

<DesignNotes>
  Search with search button is commonly used above the pages designed to manage filter pills and search while Search
  without search button is more commonly used within the pickers and modals.
</DesignNotes>
