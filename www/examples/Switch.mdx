## Switch

```jsx live=true
const Example = () => {
  const [isToggleOn, setIsToggleOn] = React.useState('partial');
  const onChange = (newValue) => setIsToggleOn(newValue);

  return (
    <>
      <div>
        <div style={{ paddingTop: 20 }}>Un-Controlled Switch without defaultChecked</div>
        <div style={{ paddingTop: 10 }}>
          <Switch />
        </div>
      </div>
      <div>
        <div style={{ paddingTop: 20 }}>Un-Controlled Switch with defaultChecked</div>
        <div style={{ paddingTop: 10 }}>
          <Switch defaultChecked={true} />
        </div>
      </div>
      <div>
        <div style={{ paddingTop: 20 }}>Un-Controlled Switch with defaultChecked with onChange</div>
        <div style={{ paddingTop: 10 }}>
          <Switch defaultChecked={false} onChange={_.noop} />
        </div>
      </div>
      <div>
        <div style={{ paddingTop: 20 }}>Controlled Switch (Partial Initial State)</div>
        <div style={{ paddingTop: 10 }}>
          <Switch checked={isToggleOn} onChange={onChange} />
        </div>
      </div>
      <div>
        <div style={{ paddingTop: 20 }}>Disabled Un-Checked Switch</div>
        <div style={{ paddingTop: 10 }}>
          <Switch disabled />
        </div>
      </div>
      <div>
        <div style={{ paddingTop: 20 }}>Disabled Checked Switch</div>
        <div style={{ paddingTop: 10 }}>
          <Switch checked disabled onChange={_.noop} />
        </div>
      </div>
    </>
  );
};

render(Example);
```
