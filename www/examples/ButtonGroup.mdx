## Button Group

```jsx live={true}
const Example = () => {
  const [isDropdownOpen, setIsDropdownOpen] = React.useState(false);

  return (
    <React.Fragment>
      <ButtonGroup color="success">
        <Button>Approve</Button>
        <Popover triggers="click" placement="bottom" popoverContent="I am a popover on click!">
          <Button
            className="svg-icon"
            aria-label="Toggle menu"
            icon={<SvgSymbol href="./assets/svg-symbols.svg#caret-down" />}
          />
        </Popover>
      </ButtonGroup>
      <ButtonGroup color="danger" variant="inverse">
        <Button>Reject</Button>
        <Popover
          triggers="disabled"
          isOpen={isDropdownOpen}
          id="popover-2"
          popoverContent="I am a Overlay on click!"
          placement="bottom"
        >
          <Button
            className="svg-icon"
            aria-label="Toggle menu"
            icon={<SvgSymbol href="./assets/svg-symbols.svg#caret-down" />}
            onClick={() => setIsDropdownOpen(!isDropdownOpen)}
          />
        </Popover>
      </ButtonGroup>
      <ButtonGroup color="primary">
        <Button>Sign off</Button>
        <Button
          className="svg-icon"
          aria-label="Toggle menu"
          icon={<SvgSymbol href="./assets/svg-symbols.svg#caret-down" />}
          onClick={() => alert('>I am a Alert on click!')}
        />
      </ButtonGroup>
      <ButtonGroup variant="inverse" disabled={true}>
        <Button>Disabled</Button>
        <Button
          className="svg-icon"
          aria-label="Toggle menu"
          icon={<SvgSymbol href="./assets/svg-symbols.svg#caret-down" />}
        />
      </ButtonGroup>
    </React.Fragment>
  );
};

render(
  <div style={{ display: 'flex' }}>
    <Example />
  </div>
);
```
