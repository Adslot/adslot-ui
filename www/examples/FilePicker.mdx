## File Picker

```jsx live=true
const FilePickerExample = () => {
  const onSelect = _.noop;
  const [fileName, setFileName] = React.useState('');

  return (
    <>
      Uncontrolled:
      <FilePicker onSelect={onSelect} />
      <br />
      Controlled:
      <FilePicker onSelect={onSelect} value={fileName} onChange={fileName => setFileName(fileName)} />
    </>
  );
};

render(
  <div style={{ width: 400 }}>
    <FilePickerExample />
  </div>
);
```
