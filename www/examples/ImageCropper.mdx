## Image Cropper

```jsx live=true
const Example = () => {
  const [data, setData] = React.useState('');
  const onCrop = value => setData(JSON.stringify(value));

  return (
    <>
      <ImageCropper src="./assets/tileGrid/example-image.jpg" onCancel={_.noop} onCrop={onCrop} height={400} />
      <br />
      <Textarea
        placeholder="Click Upload to see the output data"
        value={data}
        onChange={_.noop}
        style={{ width: 1000 }}
      />
    </>
  );
};

render(Example);
```
