## Dropdown Menu Checkbox and Radio groups

This example shows how to use Checkbox and Radio groups within a menu

```jsx live=true
const Example = () => {
  const [radio, setRadio] = React.useState('block');
  const [checkbox, setCheckbox] = React.useState([]);
  const [closeOnClick, setCloseOnClick] = React.useState(false);

  return (
    <>
      <div>Close after selection</div>
      <Switch id="switch" value={closeOnClick ? 'true' : 'false'} onChange={(v) => setCloseOnClick(v)} />
      <br />
      <p>
        Radio: {radio}
        <br />
        Checkbox: {checkbox.map((v) => v).join(', ')}
      </p>
      <DropdownMenu closeOnItemClick={closeOnClick} contentId="menu" triggerId="trigger">
        {({ closeMenu }) => (
          <>
            <DropdownMenu.Trigger variant="solid" color="primary">
              Menu
            </DropdownMenu.Trigger>
            <DropdownMenu.Content>
              <DropdownMenu.Label id="checkbox-group-label">Elements</DropdownMenu.Label>
              <DropdownMenu.CheckboxGroup
                aria-labelledby="checkbox-group-label"
                value={checkbox}
                onChange={setCheckbox}
                name="test-1"
              >
                <DropdownMenu.Checkbox value="div" label="div" />
                <DropdownMenu.Checkbox value="span" label="span" />
                <DropdownMenu.Checkbox value="section" label="section" />
                <DropdownMenu.Checkbox value="main" label="main" />
                <DropdownMenu.Checkbox value="aside" label="aside" />
              </DropdownMenu.CheckboxGroup>
              <DropdownMenu.Label id="radio-group-label">Display</DropdownMenu.Label>
              <DropdownMenu.RadioGroup
                name="test-2"
                aria-labelledby="radio-group-label"
                value={radio}
                onChange={setRadio}
              >
                <DropdownMenu.Radio value="block" label="block" />
                <DropdownMenu.Radio value="flex" label="flex" />
              </DropdownMenu.RadioGroup>
              <DropdownMenu.Divider />
              <DropdownMenu.ItemContainer>
                <Button variant="borderless" onClick={closeMenu}>
                  Cancel
                </Button>
                <Button onClick={closeMenu} color="primary">
                  Done
                </Button>
              </DropdownMenu.ItemContainer>
            </DropdownMenu.Content>
          </>
        )}
      </DropdownMenu>
    </>
  );
};

render(Example);
```
