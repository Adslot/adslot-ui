## Navigation

```jsx live=true
const Example = () => {
  const [activeKey, setActiveKey] = React.useState(0);
  onSelect = key => setActiveKey(key);

  return (
    <Nav activeKey={activeKey} onSelect={onSelect}>
      <NavItem eventKey={0}>Dashboard</NavItem>
      <NavItem eventKey={1}>Reports</NavItem>
    </Nav>
  );
};

render(Example);
```
