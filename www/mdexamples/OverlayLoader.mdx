## Overlay Loader

```jsx live=true
const Example = () => {
  const [showLoader, setShowLoader] = React.useState(false);

  const handleKeyPress = event => {
    if (event.keyCode === 27) {
      toggleDisabledLoader();
    }
  };

  const toggleDisabledLoader = () => {
    setShowLoader(!showLoader);
  };

  React.useEffect(() => {
    const eventTrigger = showLoader ? window.addEventListener : window.removeEventListener;
    eventTrigger('keydown', handleKeyPress);
  }, [showLoader]);

  return (
    <React.Fragment>
      <h4>Static markup</h4>
      <div className="aui--docs-static-markup">
        <OverlayLoader text="campaigns" />
        <OverlayLoader heading="Fetching" text="delivery data" />
      </div>
      <div className="live-demo">
        <h4>Demo</h4>
        <Button onClick={toggleDisabledLoader}>{showLoader ? 'Hide' : 'Show'} Loader (disabled background)</Button>
        {showLoader && <OverlayLoader disableBackground={true} text="Press esc to hide" />}
      </div>
    </React.Fragment>
  );
};

render(Example);
```

  Fixed position loader which provides user experience for page loading or interim loading states.
